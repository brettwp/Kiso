1 kiso.geom.ReducibleSimplePolyConvexHull = kiso.Class(
  2 	{
  3 		parent: kiso.geom.SimplePolyConvexHull,
  4 		constants: {
  5 			_POP_OPERATION_AT_HEAD: 10,
  6 			_POP_OPERATION_AT_TAIL: 11,
  7 			_PUSH_OPERATION: 12
  8 		}
  9 	},
 10 	{
 11 		_operationStack: [],
 12 
 13 		build: function() {
 14 			this._operationStack = [];
 15 			this.superclass.build();
 16 		},
 17 
 18 		getOperationStack: function() {
 19 			return this._operationStack;
 20 		},
 21 
 22 		_initializeHullIndexes: function() {
 23 			this.superclass._initializeHullIndexes();
 24 			this._operationStack.push({
 25 				index: this._hullIndexes.getHeadData(0),
 26 				operation: kiso.geom.ReducibleSimplePolyConvexHull._PUSH_OPERATION
 27 			});
 28 		},
 29 
 30 		_popHullPoint: function(atEnd) {
 31 			this._operationStack.push({
 32 				index: this.superclass._popHullPoint(atEnd),
 33 				operation:
 34 					(atEnd == kiso.geom.ReducibleSimplePolyConvexHull._AT_HEAD) ?
 35 					kiso.geom.ReducibleSimplePolyConvexHull._POP_OPERATION_AT_HEAD :
 36 					kiso.geom.ReducibleSimplePolyConvexHull._POP_OPERATION_AT_TAIL
 37 			});
 38 		},
 39 
 40 		_pushHullPoint: function(index) {
 41 			this.superclass._pushHullPoint(index);
 42 			this._operationStack.push({
 43 				index: index,
 44 				operation: kiso.geom.ReducibleSimplePolyConvexHull._PUSH_OPERATION
 45 			});
 46 		},
 47 
 48 		reduceTo: function(polyEndPoint) {
 49 			var operationEndPoint = this._findPushOperation(polyEndPoint);
 50 			this._reverseOperations(operationEndPoint);
 51 			this._reduceSimplePoly(polyEndPoint);
 52 		},
 53 
 54 		_findPushOperation: function(polyEndPoint) {
 55 			var operationEndPoint = this._operationStack.length - 1;
 56 			for (var index = operationEndPoint; index >= 0; index--) {
 57 				if (this._operationStack[index].operation ==
 58 						kiso.geom.ReducibleSimplePolyConvexHull._PUSH_OPERATION) {
 59 					operationEndPoint = index;
 60 					if (this._indexOrderInDirectionOfHull(
 61 							this._operationStack[index].index, polyEndPoint)) {
 62 						break;
 63 					}
 64 				}
 65 			}
 66 			return operationEndPoint;
 67 		},
 68 
 69 		_indexOrderInDirectionOfHull: function(index0, index1) {
 70 			return (
 71 				(
 72 					(this._direction == kiso.geom.ReducibleSimplePolyConvexHull.FIRST2LAST)
 73 						&&
 74 					(index1 >= index0)
 75 				) || (
 76 					(this._direction == kiso.geom.ReducibleSimplePolyConvexHull.LAST2FIRST)
 77 						&&
 78 					(index1 <= index0)
 79 				)
 80 			);
 81 		},
 82 
 83 		_reverseOperations: function(operationEndPoint) {
 84 			var operation;
 85 			for (var i = this._operationStack.length-1; i > operationEndPoint; i--) {
 86 				operation = this._operationStack.pop();
 87 				if (operation.operation == kiso.geom.ReducibleSimplePolyConvexHull._PUSH_OPERATION) {
 88 					this.superclass._popHullPoint(kiso.geom.ReducibleSimplePolyConvexHull._AT_HEAD);
 89 					this.superclass._popHullPoint(kiso.geom.ReducibleSimplePolyConvexHull._AT_TAIL);
 90 				} else if (operation.operation ==
 91 						kiso.geom.ReducibleSimplePolyConvexHull._POP_OPERATION_AT_HEAD) {
 92 					this.superclass._pushHullPoint(
 93 						operation.index,
 94 						kiso.geom.ReducibleSimplePolyConvexHull._AT_HEAD
 95 					);
 96 				} else if (operation.operation ==
 97 						kiso.geom.ReducibleSimplePolyConvexHull._POP_OPERATION_AT_TAIL) {
 98 					this.superclass._pushHullPoint(
 99 						operation.index,
100 						kiso.geom.ReducibleSimplePolyConvexHull._AT_TAIL
101 					);
102 				}
103 			}
104 		},
105 
106 		_reduceSimplePoly: function(polyEndPoint) {
107 			if (this._direction == kiso.geom.ReducibleSimplePolyConvexHull.LAST2FIRST) {
108 				this._simplePoly.splice(0, polyEndPoint);
109 				this._reindexHull(polyEndPoint);
110 			} else {
111 				this._simplePoly.splice(polyEndPoint+1);
112 			}
113 		},
114 
115 		_reindexHull: function(polyEndPoint) {
116 			if (this._direction == kiso.geom.ReducibleSimplePolyConvexHull.LAST2FIRST) {
117 				var iterator = new kiso.data.ListIterator(this._hullIndexes);
118 				while(1) {
119 					iterator.setData(iterator.getData() - polyEndPoint);
120 					if (iterator.hasNext()) {
121 						iterator.gotoNext();
122 					} else {
123 						break;
124 					}
125 				}
126 			}
127 		}
128 	}
