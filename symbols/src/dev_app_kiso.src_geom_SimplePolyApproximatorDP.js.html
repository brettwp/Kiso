 14 		},
 15 
 16 		setPoints: function(simplePoly) {
 17 			this._simplePoly = simplePoly.map(function(point) {return point.clone();});
 18 		},

 20 		build: function() {
 21 			this._initializeSections();
 22 			this._iterator = new kiso.data.ListIterator(this._subSections);
 23 			var doNext = true;
 24 			while(doNext) {
 25 				this._simplifyCurrentSection();
 26 				doNext = this._iterator.hasNext();
 27 				if (doNext) this._iterator.gotoNext();
 28 			}
 29 		},
 30 
 31 		_initializeSections: function() {
 32 			var section = this._newSection();
 33 			section.firstPoint = 0;
 34 			section.lastPoint = this._simplePoly.length-1;
 35 			this._subSections = new kiso.data.LinkedList();
 36 			this._subSections.addFirst(section);
 37 		},
 38 
 39 		_simplifyCurrentSection: function() {
 40 			var section = this._iterator.getData();
 41 			while (section.distanceSquared == null) {
 42 				this._findFarthestInCurrentSection();
 43 				if (section.distanceSquared > this._stopToleranceSquared) {
 44 					this._seperateCurrentSection();
 45 				}
 46 				section = this._iterator.getData();
 47 			}
 48 		},
 49 
 50 		_seperateCurrentSection: function() {
 51 			var sectionLeft = this._iterator.getData();
 52 			var sectionRight = this._newSection();
 53 			sectionRight.firstPoint = sectionLeft.farthestPoint;
 54 			sectionRight.lastPoint = sectionLeft.lastPoint;
 55 			sectionLeft.lastPoint = sectionLeft.farthestPoint;
 56 			sectionLeft.farthestPoint = null;
 57 			sectionLeft.distanceSquared = null;
 58 			this._iterator.addAfter(sectionRight);
 59 		},
 60 
 61 		_findFarthestInCurrentSection: function() {
 62 			var section = this._iterator.getData();
 63 			var point0 = this._simplePoly[section.firstPoint];
 64 			var point1 = this._simplePoly[section.lastPoint];
 65 			var deltaX = point1.getX() - point0.getX();
 66 			var deltaY = point1.getY() - point0.getY();
 67 			var numeratorMax = 0;
 68 			var numerator, indexMax;
 69 			for (var index = section.firstPoint+1; index < section.lastPoint; index++) {
 70 				numerator = Math.abs(
 71 					(deltaX * (point0.getY() - this._simplePoly[index].getY())) +
 72 					(deltaY * (this._simplePoly[index].getX() - point0.getX()))
 73 				);
 74 				if (numerator > numeratorMax) {
 75 					numeratorMax = numerator;
 76 					indexMax = index;
 77 				}
 78 			}
 79 			section.farthestPoint = indexMax;
 80 			section.distanceSquared = numeratorMax*numeratorMax/(deltaX*deltaX + deltaY*deltaY);
 81 		},
 82 
 83 		_newSection: function() {
 84 			return {
 85 				firstPoint: null,
 86 				lastPoint: null,
 87 				farthestPoint: null,
 88 				distanceSquared: null
 89 			};
 90 		},
 91 
 92 		setTolerance: function(tolerance) {
 93 			this._stopTolerance = tolerance;
 94 			this._stopToleranceSquared = tolerance*tolerance;
 95 		},
 96

 99 		},
100 
101 		getIndexes: function() {

103 			var doNext = true;
104 			var sectionArray = [];

107 				doNext = iterator.hasNext();
108 				if (doNext) iterator.gotoNext();
109 			}
110 			sectionArray.push(iterator.getData().lastPoint);
111 			return sectionArray;
112 		}
