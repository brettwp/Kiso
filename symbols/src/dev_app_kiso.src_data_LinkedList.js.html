1 /**
  2  * @class
  3  * @description A linked list.
  4  * @augments kiso.data.AbstractList
  5  */
  6 kiso.data.LinkedList = kiso.Class(
  7   {
  8     parent: kiso.data.AbstractList,
  9     interfaces: kiso.data.ILinkedList
 10   },
 11   /** @lends kiso.data.LinkedList.prototype */
 12   {
 13     /**
 14      * @description Add a new element to the front of the list.
 15      * @param data The data for the new first element.
 16      */
 17     addFirst: function(data) {
 18       this._addAfter(this._first, data);
 19     },
 20 
 21     /**
 22      * @description Add a new element to the end of the list.
 23      * @param data The data for the new last element.
 24      */
 25     addLast: function(data) {
 26       this._addBefore(this._last, data);
 27     },
 28 
 29     /**
 30      * @description Add a new element just before the indexed element.
 31      * @param {Integer} index The zero based offset from the front of the list.
 32      * @param data The data for the new element.
 33      */
 34     addBefore: function(index, data) {
 35       this._addBefore(this._getNode(index), data);
 36     },
 37 
 38     /**
 39      * @description Add a new element just after the indexed element.
 40      * @param {Integer} index The zero based offset from the front of the list.
 41      * @param data The data for the new element.
 42      */
 43     addAfter: function(index, data) {
 44       this._addAfter(this._getNode(index), data);
 45     },
 46 
 47     /**
 48      * @description Remove the specfied element.
 49      * @param {Integer} index The zero based offset from the front of the list.
 50      * @throws A an error if the index is out of bounds.
 51      */
 52     remove: function(index) {
 53       return this._removeNode(this._getNode(index));
 54     },
 55 
 56     /**
 57      * @description Remove the first element in the list.
 58      * @throws A an error if the list is empty.
 59      */
 60     removeFirst: function() {
 61       if (this.isEmpty()) {
 62         throw new Error('Cannot remove first from empty LinkedList');
 63       }
 64       return this._removeNode(this._first._next);
 65     },
 66 
 67     /**
 68      * @description Remove the last element in the list.
 69      * @throws A an error if the list is empty.
 70      */
 71     removeLast: function() {
 72       if (this.isEmpty()) {
 73         throw new Error('Cannot remove last from empty LinkedList');
 74       }
 75       return this._removeNode(this._last._prev);
 76     },
 77 
 78     /**
 79      * @description Get the data of an element without altering the list.
 80      * @param {Integer} index The zero base offset from the front of the list.
 81      * @returns The data from the specified element.
 82      * @throws A an error if the list is empty.
 83      */
 84     getData: function(index) {
 85       return this._getNode(index)._data;
 86     }
 87   }
 88 );
