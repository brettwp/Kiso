1 /**
  2  * @description Creates a new class.
  3  * @param {Class|Object} parentClassOrObj The parent class, an object of parameters (see paramObj) or the class definition object (see classObj).
  4  * @param {Object} childDefinition The class definition object (see classObj) when the first arguments is an object.
  5  *
  6  * There are three basic ways to instantiate a class:
  7  * <ol>
  8  *   <li><code>Class(classObject)</code> - A base class.</li>
  9  *   <li><code>Class(parentClass, classObject)</code> - A sub-class that inherits from a parent class.</li>
 10  *   <li><code>Class(paramObject, classObject)</code> - A base or sub-class that uses a paramObject.</li>
 11  * </ol>
 12  * A <code>parentClass</code> is any class created by a previous call to <code>kiso.Class</code> where the
 13  * <code>paramObject</code> and <code>classObject</code> are defined as follows.
 14  *
 15  * <dl>
 16  *   <dt><strong><code>paramObject</code></strong><dt>
 17  *   <dd>Contains atleast one of the following:
 18  * <pre>{
 19  *   parent: parentClass,
 20  *   interfaces: [a single interface or an array of interfaces],
 21  *   constants: {an object of constant names and values}
 22  * }</pre>
 23  *     <p>Note that constants are attached to the class.  For example a class defined like,<p>
 24  *     <pre>  var example = kiso.Class({ constants: { ONE:1, TWO:2 } }, {...});</pre>
 25  *     <p>the constants will be</p>
 26  *     <pre>
 27  *   example.ONE == 1
 28  *   example.TWO == 2
 29  *     </pre>
 30  *   </dd>
 31  *
 32  *   <dt><strong><code>classObject</code></strong></dt>
 33  *   <dd>An object of variables and methods (using <code>initialize</code> for the constructor):
 34  * <pre>{
 35  *   variable1: 1,
 36  *   variable2: 2,
 37  *   initialize: funciton(...) {...},
 38  *   method1: funciton(...) {...},
 39  *   method2: funciton(...) {...}
 40  * }</pre>
 41  *   </dd>
 42  * </dl>
 43  *
 44  * @returns {Class} A new class
 45  */
 46 kiso.Class = function(parentClassOrObj, childDefinition) {
 47   return create(parentClassOrObj, childDefinition);
 48 
 49   function create(parentClassOrObj, childDefinition) {
 50     var interfaces = null;
 51     var constants = null;
 52     var parentClass = null;
 53     if (childDefinition == undefined) {
 54       childDefinition = parentClassOrObj;
 55     } else if (typeof parentClassOrObj == 'object') {
 56       if (parentClassOrObj.hasOwnProperty('parent') && !parentClassOrObj.parent) {
 57         throw new Error('Parent class undefined.');
 58       } else {
 59         parentClass = parentClassOrObj.parent;
 60       }
 61       if (parentClassOrObj.hasOwnProperty('interfaces') && !parentClassOrObj.interfaces) {
 62         throw new Error('Interface undefined.');
 63       } else {
 64         interfaces = parentClassOrObj.interfaces;
 65       }
 66       constants = parentClassOrObj.constants;
 67     } else if (parentClassOrObj) {
 68       parentClass = parentClassOrObj;
 69     } else {
 70       throw new Error('Parent class undefined.');
 71     }
 72     var newClass = function() {
 73       createUniqueInstanceVariables(this);
 74       if (this.__superclass) this.superclass = new this.__superclass(this);
 75       if (this.initialize) this.initialize.apply(this, arguments);
 76     };
 77     setupClassFromParent(newClass, parentClass);
 78     extendClassMembers(newClass, childDefinition);
 79     extendClassInterfaces(newClass, interfaces);
 80     ensureImplementsInterfaces(newClass);
 81     setupClassConstants(newClass, constants);
 82     return newClass;
 83   }
 84 
 85   function createUniqueInstanceVariables(obj) {
 86     for (var prop in obj) {
 87       if (prop != '__superclass') {
 88         if (obj[prop] != null &&
 89           typeof obj[prop] == 'object') obj[prop] = clone(obj[prop]);
 90       }
 91     }
 92   }
 93 
 94   function clone(oldObj) {
 95     var newObj = (oldObj instanceof Array) ? [] : {};
 96     for (var prop in oldObj) {
 97       if (typeof oldObj[prop] == 'object') {
 98         newObj[prop] = clone(oldObj[prop]);
 99       } else {
100         newObj[prop] = oldObj[prop];
101       }
102     }
103     return newObj;
104   }
105 
106   function setupClassFromParent(newClass, parentClass) {
107     if (parentClass) {
108       var func = function() {};
109       func.prototype = parentClass.prototype;
110       newClass.prototype = new func();
111       newClass.prototype.__superclass = function(subObject) {
112         this._subObject = subObject;
113         this._parentClass = parentClass.prototype;
114       };
115       setupClassConstants(newClass, parentClass);
116     }
117   }
118 
119   function extendClassMembers(newClass, extension) {
120     var extObj, prop;
121     if (typeof extension == 'function') {
122       extObj = extension.prototype;
123     } else {
124       extObj = extension
125     }
126     for (prop in extObj) {
127       if (newClass.prototype.__superclass &&
128           newClass.prototype[prop] && typeof newClass.prototype[prop] == 'function') {
129         newClass.prototype.__superclass.prototype[prop] = wrapParentFunction(prop);
130       }
131       newClass.prototype[prop] = extObj[prop];
132     }
133   }
134 
135   function wrapParentFunction(superFunc) {
136     var sFunc = superFunc;
137     return function() { return this._parentClass[sFunc].apply(this._subObject,arguments) }
138   }
139 
140   function extendClassInterfaces(newClass, interfaces) {
141     if (interfaces) {
142       interfaces = (interfaces instanceof Array) ? interfaces : [interfaces];
143       newClass.__interfaces = newClass.__interfaces || [];
144       for (var index in interfaces) {
145         newClass.__interfaces.push(interfaces[index]);
146       }
147     }
148   }
149 
150   function ensureImplementsInterfaces(testClass) {
151     if (testClass.__interfaces) {
152       var index;
153       var methods = [];
154       for (index in testClass.__interfaces) {
155         methods = methods.concat(testClass.__interfaces[index].getMethods());
156       }
157       for (index in methods) {
158         if (typeof testClass.prototype[methods[index]] != 'function') {
159           throw new Error('Class does not implement interface(s).');
160         }
161       }
162     }
163   }
164 
165   function setupClassConstants(newClass, constants) {
166     if (constants) {
167       for (var constKey in constants) {
168         newClass[constKey] = constants[constKey];
169       }
170     }
171   }
172 };
