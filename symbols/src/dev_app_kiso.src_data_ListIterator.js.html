1 kiso.data.ListIterator = kiso.Class(
  2   {
  3     constants: /** @lends kiso.data.ListIterator */ {
  4       /** @constant */
  5       STARTATFIRST: 0,
  6       /** @constant */
  7       STARTATLAST: 1
  8     }
  9   },
 10   /** @lends kiso.data.ListIterator.prototype */
 11   {
 12     _list: null,
 13     _currentNode: null,
 14     _index: null,
 15 
 16     /**
 17      * @constructs
 18      * @description Creates a new iterator for the given list and in the specified direction.
 19      * @param {List} list The list to iterator over.
 20      * @param {STARTATFIRST|STARTATLAST} direction Which end of the list to start from.
 21      * @returns {ListIterator} A new list iterator.
 22      */
 23     initialize: function(list, direction) {
 24       if (list.isEmpty()) {
 25         throw new Error('Cannot iterate on empty list');
 26       }
 27       this._list = list;
 28       this._currentNode = (direction == kiso.data.ListIterator.STARTATLAST) ?
 29         this._list._last._prev : this._list._first._next;
 30       this._index = (direction == kiso.data.ListIterator.STARTATLAST) ?
 31         this._list._size - 1 : 0;
 32     },
 33 
 34     /**
 35      * @description Adds a new element to the list after the current element.
 36      * @param data The data of the new element.
 37      */
 38     addAfter: function(data) {
 39       this._list._addAfter(this._currentNode, data);
 40     },
 41 
 42     /**
 43      * @description Adds a new element to the list before the current element.
 44      * @param data The data of the new element.
 45      */
 46     addBefore: function(data) {
 47       this._list._addBefore(this._currentNode, data);
 48       this._index++;
 49     },
 50 
 51     /**
 52      * @description Returns whether there is a next element.
 53      * @returns True if the current element has an element after it.
 54      */
 55     hasNext: function() {
 56       return (this._currentNode._next != this._list._last);
 57     },
 58 
 59     /**
 60      * @description Returns whether there is a previous element.
 61      * @returns True if the current element has an element before it.
 62      */
 63     hasPrevious: function() {
 64       return (this._currentNode._prev != this._list._first);
 65     },
 66 
 67     /**
 68      * @description Get the index of the current element.
 69      * @returns The zero based index from the front of the list of the current element.
 70      */
 71     getIndex: function() {
 72       return this._index;
 73     },
 74 
 75     /**
 76      * @description Moves the ListIterator to the next element in the list.
 77      * @throws An error if there is no next element.
 78      */
 79     gotoNext: function() {
 80       if (this.hasNext()) {
 81         this._currentNode = this._currentNode._next;
 82         this._index++;
 83       } else {
 84         throw new Error('No such element');
 85       }
 86     },
 87 
 88     /**
 89      * @description Moves the ListIterator to the previous element in the list.
 90      * @throws An error if there is no previous element.
 91      */
 92     gotoPrevious: function() {
 93       if (this.hasPrevious()) {
 94         this._currentNode = this._currentNode._prev;
 95         this._index--;
 96       } else {
 97         throw new Error('No such element');
 98       }
 99     },
100 
101     /**
102      * @description Removes the current element from the list.
103      */
104     remove: function() {
105       var oldNode = this._currentNode;
106       if (this.hasNext()) {
107         this.gotoNext();
108       } else if (this.hasPrevious()) {
109         this.gotoPrevious();
110       } else {
111         this._currentNode = null;
112       }
113       this._list._removeNode(oldNode);
114     },
115 
116     /**
117      * @description Gets the data of the current element without altering the list.
118      * @returns The data for the current element.
119      */
120     getData: function() {
121       return this._currentNode._data;
122     },
123 
124     /**
125      * @description Sets the data of the current element.
126      * @param data The new data for the current element.
127      */
128     setData: function(data) {
129       this._currentNode._data = data;
130     }
131   }
