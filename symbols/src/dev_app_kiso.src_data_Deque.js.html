1 /**
  2  * @class
  3  * @description A double ended queue
  4  * @augments kiso.data.AbstractList
  5  */
  6 kiso.data.Deque = kiso.Class(
  7   {
  8     parent: kiso.data.AbstractList,
  9     interfaces: kiso.data.IDeque
 10   },
 11   /** @lends kiso.data.Deque.prototype */
 12   {
 13     /**
 14      * @description Push a new element on to the head of the list.
 15      * @param data The data for the new element.
 16      */
 17     pushHead: function(data) {
 18       this._addBefore(this._last, data);
 19     },
 20 
 21     /**
 22      * @description Push a new element on to the tail of the list.
 23      * @param data The data for the new element.
 24      */
 25     pushTail: function(data) {
 26       this._addAfter(this._first, data);
 27     },
 28 
 29     /**
 30      * @description Pop the last element off the head of the list.
 31      * @returns The data of the last element.
 32      * @throws A error when the list is empty.
 33      */
 34     popHead: function() {
 35       if (this.isEmpty()) {
 36         throw new Error('Cannot pop head on empty Deque');
 37       }
 38       return this._removeNode(this._last._prev);
 39     },
 40 
 41     /**
 42      * @description Pop the last element off the tail of the list.
 43      * @returns The data of the last element.
 44      * @throws A error when the list is empty.
 45      */
 46     popTail: function() {
 47       if (this.isEmpty()) {
 48         throw new Error('Cannot pop tail on empty Deque');
 49       }
 50       return this._removeNode(this._first._next);
 51     },
 52 
 53     /**
 54      * @description Retrieves the data of the indexed element starting at the head of the list.  Does not alter the list.
 55      * @param {integer} index The element index (zero based) from the head of the list.
 56      * @returns The data from the specified element.
 57      *
 58      */
 59     getHeadData: function(index) {
 60       return this._getNode(this._size-index-1)._data;
 61     },
 62 
 63     /**
 64      * @description Retrieves the data of the indexed element starting at the tail of the list.  Does not alter the list.
 65      * @param {integer} index The element index (zero based) from the head of the list.
 66      * @returns The data from the specified element.
 67      *
 68      */
 69     getTailData: function(index) {
 70       return this._getNode(index)._data;
 71     }
 72   }
 73 );
