1 kiso.geom.SimplePolyConvexHull = kiso.Class(
  2 	{
  3 		interfaces: kiso.geom.IConvexHull,
  4 		constants: {
  5 			_AT_HEAD: 0,
  6 			_AT_TAIL: 1,
  7 			FIRST2LAST: 2,
  8 			LAST2FIRST: 3
  9 		}
 10 	},
 11 	{
 12 		_simplePoly: null,
 13 		_hullIndexes: null,
 14 		_direction: null,
 15 
 16 		initialize: function(simplePoly, direction) {
 17 			this.setPoints(simplePoly);
 18 			this.setDirection(direction);
 19 		},
 20 
 21 		setPoints: function(simplePoly) {
 22 			this._simplePoly = simplePoly.map(function(point) { return point.clone(); });
 23 			this._hullIndexes = null;
 24 		},
 25 
 26 		getPoints: function() {
 27 			return this._simplePoly;
 28 		},
 29 
 30 		setDirection: function(direction) {
 31 			this._direction =
 32 				(direction == kiso.geom.SimplePolyConvexHull.LAST2FIRST) ?
 33 				kiso.geom.SimplePolyConvexHull.LAST2FIRST :
 34 				kiso.geom.SimplePolyConvexHull.FIRST2LAST;
 35 		},
 36 
 37 		getHullIndexes: function() {
 38 			return (this._hullIndexes) ? this._hullIndexes.toArray() : null;
 39 		},
 40 
 41 		build: function() {
 42 			if (this._simplePoly && this._simplePoly.length >= 3) {
 43 				this._initializeHullIndexes();
 44 				this._expandHull();
 45 			}
 46 		},
 47 
 48 		_initializeHullIndexes: function() {
 49 			this._hullIndexes = new kiso.data.Deque();
 50 			var indexes, point0, point1, point2;
 51 			if (this._direction == kiso.geom.SimplePolyConvexHull.LAST2FIRST) {
 52 				point0 = this._simplePoly.length-1;
 53 				point1 = point0-1;
 54 				point2 = point1-1;
 55 			} else {
 56 				point0 = 0;
 57 				point1 = 1;
 58 				point2 = 2;
 59 			}
 60 			if (this._simplePoly[point2].isLeftOfVector(
 61 					this._simplePoly[point0], this._simplePoly[point1]
 62 				)) {
 63 				indexes = [point2, point0, point1, point2];
 64 			} else {
 65 				indexes = [point2, point1, point0, point2];
 66 			}
 67 			for (var i = 0; i < 4; i++) {
 68 				this._hullIndexes.pushHead(indexes[i]);
 69 			}
 70 		},
 71 
 72 		_expandHull: function() {
 73 			var index, increment;
 74 			if (this._direction == kiso.geom.SimplePolyConvexHull.LAST2FIRST) {
 75 				index = this._simplePoly.length-4;
 76 				increment = -1;
 77 			} else {
 78 				index = 3;
 79 				increment = 1;
 80 			}
 81 			for (; (0 <= index && index < this._simplePoly.length); index += increment) {
 82 				this._maintainConvexity(index, kiso.geom.SimplePolyConvexHull._AT_HEAD);
 83 				this._maintainConvexity(index, kiso.geom.SimplePolyConvexHull._AT_TAIL);
 84 			}
 85 		},
 86 
 87 		_maintainConvexity: function(index, atEnd) {
 88 			var pushCurrentPoint = false;
 89 			var popHullPoint = true;
 90 			while (popHullPoint) {
 91 				popHullPoint = this._isPointLeftOfVector(index, atEnd);
 92 				if (popHullPoint) {
 93 					this._popHullPoint(atEnd);
 94 					pushCurrentPoint = true;
 95 				}
 96 			}
 97 			if (pushCurrentPoint) {
 98 				this._pushHullPoint(index);
 99 			}
100 		},
101 
102 		_isPointLeftOfVector: function(index, atEnd) {
103 			var atHead = (atEnd == kiso.geom.SimplePolyConvexHull._AT_HEAD);
104 			var index0 = atHead ? this._hullIndexes.getHeadData(1) : this._hullIndexes.getTailData(0);
105 			var index1 = atHead ? this._hullIndexes.getHeadData(0) : this._hullIndexes.getTailData(1);
106 			return !this._simplePoly[index].isLeftOfVector(
107 				this._simplePoly[index0], this._simplePoly[index1]
108 			);
109 		},
110 
111 		_popHullPoint: function(atEnd) {
112 			if (atEnd == kiso.geom.SimplePolyConvexHull._AT_HEAD) {
113 				return this._hullIndexes.popHead()
114 			}	else {
115 				return this._hullIndexes.popTail();
116 			}
117 		},
118 
119 		_pushHullPoint: function(index, atEnd) {
120 			if (atEnd != kiso.geom.SimplePolyConvexHull._AT_HEAD) this._hullIndexes.pushTail(index);
121 			if (atEnd != kiso.geom.SimplePolyConvexHull._AT_TAIL) this._hullIndexes.pushHead(index);
122 		}
123 	}
