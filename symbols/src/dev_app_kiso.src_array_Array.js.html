133   _useNativeOrWrapper: function(func, args) {
134     if (Array.prototype[func]) {
135       var theArray = Array.prototype.shift.call(args);
136       return Array.prototype[func].apply(theArray, args);
137     } else {
138       return this['_'+func].apply(this, args);
139     }
140   },

142   _every: function(theArray, callbackFunction, thisContext) {
143     if (typeof callbackFunction !== "function") throw new TypeError();
144     var length = theArray.length;
145     for (var index = 0; index < length; index++) {
146       if (index in theArray) {
147         if (!callbackFunction.call(thisContext, theArray[index], index, theArray)) break;
148       }
149     }
150     return (index == length);
151   },
152 
153   _forEach: function(theArray, callbackFunction, thisContext) {
154     if (typeof callbackFunction !== "function") throw new TypeError();
155     var length = theArray.length;
156     for (var index = 0; index < length; index++) {
157       callbackFunction.call(thisContext, theArray[index], index, theArray);
158     }
159   },
160 
161   _filter: function(theArray, callbackFunction, thisContext) {
162     if (typeof callbackFunction !== "function") throw new TypeError();
163     var filteredArray = [];
164     var length = theArray.length;
165     for (var index = 0; index < length; index++) {
166       var valueBeforeCall = theArray[index];
167       if (callbackFunction.call(thisContext, theArray[index], index, theArray))
168         filteredArray.push(valueBeforeCall);
169     }
170     return filteredArray;
171   },

173   _indexOf: function(theArray, searchElement, fromIndex) {

181   _findIndexOf: function(theArray, searchElement, fromIndex, direction) {
182     var length = theArray.length;
183     var index = (typeof fromIndex == 'undefined') ? 0 : (
184       (fromIndex >= 0) ? fromIndex : Math.max(length + fromIndex, 0)
185     );
186     while(0 <= index && index < length) {
187       if (index in theArray && theArray[index] === searchElement) return index;
188       index += direction;
189     }
190     return -1;
191   },

193   _map: function(theArray, callbackFunction, thisContext) {
194     if (typeof callbackFunction !== "function") throw new TypeError();
195     var length = theArray.length;
196     var mappedArray = new Array(length);
197     for (var index = 0; index < length; index++) {
198       mappedArray[index] = callbackFunction.call(thisContext, theArray[index], index, theArray);
199     }
200     return mappedArray;
201   },

203   _some: function(theArray, callbackFunction, thisContext) {

206     for (var index = 0; index < length; index++) {
207       if (callbackFunction.call(thisContext, theArray[index], index, theArray)) break;
208     }
209     return (index != length);
210   },

220   _reduceFromDirection: function(theArray, callbackFunction, initialValue, direction) {
221     var length = theArray.length;
222     var accumulator;
223     var index = (direction == -1) ? length-1 : 0;
224     //initialValue = (typeof initialValue != 'undefined') ? initialValue : 0;
225     if (typeof initialValue != 'undefined') {
226       accumulator = initialValue;
227     } else {
228       do {
229         if (index in theArray) break;
230         index += direction;
231         if (index < 0 || index >= length) throw new TypeError();
232       } while(true);
233       accumulator = theArray[index];
234       index += direction;
235     }
236     while (0 <= index && index < length) {
237       if (index in theArray)
238         accumulator = callbackFunction.call(
239           undefined, accumulator, theArray[index], index, theArray
240         );
241       index += direction;
242     }
243     return accumulator;
244   }
