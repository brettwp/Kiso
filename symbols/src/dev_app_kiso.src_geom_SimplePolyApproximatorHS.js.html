1 kiso.geom.SimplePolyApproximatorHS = kiso.Class(
  2 	{
  3 		parent: kiso.geom.SimplePolyApproximatorDP
  4 	},
  5 	{
  6 		_initializeSections: function() {
  7 			this.superclass._initializeSections();
  8 			var section = this._subSections.getData();
  9 			this._updateSectionHulls(section);
 10 		},
 11 
 12 		_seperateCurrentSection: function() {
 13 			this.superclass._seperateCurrentSection();
 14 			var sectionLeft, sectionRight;
 15 			sectionLeft = this._iterator.getData();
 16 			this._iterator.gotoNext();
 17 			sectionRight = this._iterator.getData();
 18 			this._iterator.gotoPrevious();
 19 			sectionRight.lastToFirstHull = sectionLeft.lastToFirstHull;
 20 			sectionLeft.lastToFirstHull = null;
 21 			this._updateSectionHulls(sectionLeft);
 22 			this._updateSectionHulls(sectionRight);
 23 		},
 24 
 25 		_updateSectionHulls: function(section) {
 26 			var sectionSize = section.lastPoint - section.firstPoint;
 27 			if (sectionSize > 6) {
 28 				if (section.firstToLastHull) {
 29 					section.firstToLastHull.reduceTo(sectionSize);
 30 				} else {
 31 					section.firstToLastHull = new kiso.geom.ReducibleSimplePolyConvexHull(
 32 						this._simplePoly.slice(section.firstPoint, section.lastPoint+1)
 33 					);
 34 					section.firstToLastHull.build();
 35 				}
 36 				if (section.lastToFirstHull) {
 37 					section.lastToFirstHull.reduceTo(section.lastPoint - section.firstPoint);
 38 				} else {
 39 					section.lastToFirstHull = new kiso.geom.ReducibleSimplePolyConvexHull(
 40 						this._simplePoly.slice(section.firstPoint, section.lastPoint+1),
 41 						kiso.geom.ReducibleSimplePolyConvexHull.LAST2FIRST
 42 					);
 43 					section.lastToFirstHull.build();
 44 				}
 45 			} else {
 46 				section.lastToFirstHull = null;
 47 				section.firstToLastHull = null;
 48 			}
 49 		},
 50 
 51 		_findFarthestInCurrentSection: function() {
 52 			var section = this._iterator.getData();
 53 			if ((section.lastPoint - section.firstPoint) > 6) {
 54 				this._useHullToFindFarthest();
 55 			} else {
 56 				this.superclass._findFarthestInCurrentSection();
 57 			}
 58 		},
 59 
 60 		_useHullToFindFarthest: function() {
 61 			var section = this._iterator.getData();
 62 			var point0 = this._simplePoly[section.firstPoint];
 63 			var point1 = this._simplePoly[section.lastPoint];
 64 			var hullData = {
 65 				point0: point0,
 66 				point1: point1,
 67 				vectorX: point1.getX() - point0.getX(),
 68 				vectorY: point1.getY() - point0.getY(),
 69 				hullIndexes: section.firstToLastHull.getHullIndexes(),
 70 				distanceSquared: null,
 71 				farthestPoint: null
 72 			};
 73 			for (var i=0; i<hullData.hullIndexes.length; i++) {
 74 				hullData.hullIndexes[i] += section.firstPoint;
 75 			}
 76 			this._bruteSearchHull(hullData);
 77 			if (hullData.hullIndexes.length > 6) {
 78 				this._binarySearchHull(hullData);
 79 			} else {
 80 				this._bruteSearchHull(hullData);
 81 			}
 82 			section.farthestPoint = hullData.farthestPoint;
 83 			section.distanceSquared = hullData.distanceSquared;
 84 		},
 85 
 86 		_binarySearchHull: function(hullData) {
 87 			var edgeLow, edgeHigh, edgeMid, edgeFar1, edgeFar2, slopeSignBase, slopeSignBreak;
 88 			edgeLow = 0;
 89 			edgeHigh = hullData.hullIndexes.length-1;
 90 			slopeSignBase = this._computeSlopeSign(hullData, edgeLow, edgeLow+1);
 91 			do {
 92 				edgeMid = Math.floor((edgeLow + edgeHigh)/2);
 93 				slopeSignBreak = this._computeSlopeSign(hullData, edgeMid, edgeMid+1);
 94 				if (slopeSignBase == slopeSignBreak) {
 95 					if (slopeSignBase == this._computeSlopeSign(hullData, edgeLow, edgeMid+1)) {
 96 						edgeLow = edgeMid + 1;
 97 					} else {
 98 						edgeHigh = edgeMid;
 99 					}
100 				}
101 			} while (slopeSignBase == slopeSignBreak);
102 			edgeFar1 = edgeMid;
103 			edgeFar2 = edgeMid;
104 			while (edgeLow < edgeFar1) {
105 				edgeMid = Math.floor((edgeLow + edgeFar1)/2);
106 				if (slopeSignBase == this._computeSlopeSign(hullData, edgeMid, edgeMid+1)) {
107 					edgeLow = edgeMid + 1;
108 				} else {
109 					edgeFar1 = edgeMid;
110 				}
111 			}
112 			while (edgeFar2 < edgeHigh) {
113 				edgeMid = Math.floor((edgeFar2 + edgeHigh)/2);
114 				if (slopeSignBase == this._computeSlopeSign(hullData, edgeMid, edgeMid+1)) {
115 					edgeHigh = edgeMid;
116 				} else {
117 					edgeFar2 = edgeMid + 1;
118 				}
119 			}
120 			edgeFar2--;
121 			this._compareFarthestCandidate(hullData, edgeFar1);
122 			this._compareFarthestCandidate(hullData, edgeFar2);
123 		},
124 
125 		_computeSlopeSign: function(hullData, index0, index1) {
126 			var point0 = this._simplePoly[hullData.hullIndexes[index0]];
127 			var point1 = this._simplePoly[hullData.hullIndexes[index1]];
128 			var dotProd = hullData.vectorX*(point1.getX() - point0.getX()) +
129 				hullData.vectorY*(point1.getY() - point0.getY());
130 			return dotProd < 0 ? -1 : 1;
131 		},
132 
133 		_bruteSearchHull: function(hullData) {
134 			for (var i=0; i<hullData.hullIndexes.length; i++) {
135 				this._compareFarthestCandidate(hullData, i);
136 			}
137 		},
138 
139 		_compareFarthestCandidate: function(hullData, edgeIndex) {
140 			var distanceSquared = this._simplePoly[hullData.hullIndexes[edgeIndex]]
141 				.distanceSquaredToLine(hullData.point0, hullData.point1);
142 			if (hullData.distanceSquared == null ||
143 					(distanceSquared > hullData.distanceSquared) ||
144 					(
145 						(distanceSquared == hullData.distanceSquared) &&
146 						(hullData.hullIndexes[edgeIndex] < hullData.farthestPoint)
147 					)) {
148 				hullData.distanceSquared = distanceSquared;
149 				hullData.farthestPoint = hullData.hullIndexes[edgeIndex];
150 			}
151 		},
152 
153 		_newSection: function() {
154 			return {
155 				firstPoint: null,
156 				lastPoint: null,
157 				farthestPoint: null,
158 				distanceSquared: null,
159 				firstToLastHull: null,
160 				lastToFirstHull: null
161 			};
162 		}
